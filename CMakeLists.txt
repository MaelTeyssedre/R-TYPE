project(RTYPE)
cmake_minimum_required(VERSION 3.10)
if (UNIX)
    add_definitions("-W -Wall -Wextra -fPIC")
endif (UNIX)
set(CMAKE_AUTOMOC ON)
set(CMAKE_AUTORCC ON)
set(CMAKE_AUTOUIC ON)
set(CONAN_DISABLE_CHECK_COMPILER TRUE CACHE INTERNAL "")
include (${CMAKE_BINARY_DIR}/conanbuildinfo.cmake)
conan_basic_setup(KEEP_RPATHS)
file(GLOB_RECURSE SOURCES_SERVER ${PROJECT_SOURCE_DIR}/Sources/Server/*.hpp ${PROJECT_SOURCE_DIR}/Sources/Server/*.cpp)
file(GLOB_RECURSE SOURCES_CLIENT ${PROJECT_SOURCE_DIR}/Sources/Client/*.hpp ${PROJECT_SOURCE_DIR}/Sources/Client/*.cpp)
add_executable(babel_client ${SOURCES_CLIENT})
add_executable(babel_server ${SOURCES_SERVER})
target_include_directories(rtype_client PUBLIC ${PROJECT_SOURCE_DIR}/Sources/Client/Headers/)
target_include_directories(rtype_server PUBLIC ${PROJECT_SOURCE_DIR}/Sources/Client/Headers/)
target_link_libraries(babel_client ${CONAN_LIBS})
target_link_libraries(babel_server ${CONAN_LIBS})
